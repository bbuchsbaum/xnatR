    **1. xnat_issue_token() (exported, AUTH-05):**

    ```r
    #' Issue an Alias Token
    #'
    #' Generates a new alias token from the authenticated XNAT server.
    #' Alias tokens can be used as temporary credentials with limited lifetime.
    #' Requires an active authenticated session.
    #'
    #' @return A list with components: alias (character), secret (character),
    #'   expires (character - ISO timestamp of estimated expiration).
    #' @export
    xnat_issue_token <- function() {
      response <- xnat_request("GET", "/data/services/tokens/issue",
                               accept = "application/json")
      token_data <- xnat_parse_response(response, as = "json")
      list(
        alias = token_data$alias,
        secret = token_data$secret,
        expires = token_data$estimatedExpirationTime
      )
    }
    ```

    **2. xnat_validate_token() (exported, AUTH-06):**

    ```r
    #' Validate an Alias Token
    #'
    #' Checks whether a previously issued alias token is still valid.
    #'
    #' @param alias Character. The alias portion of the token.
    #' @param secret Character. The secret portion of the token.
    #'
    #' @return Logical. TRUE if the token is valid, FALSE otherwise.
    #' @export
    xnat_validate_token <- function(alias, secret) {
      if (missing(alias) || missing(secret)) {
        stop("Both 'alias' and 'secret' must be provided.", call. = FALSE)
      }
      path <- sprintf("/data/services/tokens/validate/%s/%s",
                       utils::URLencode(alias, reserved = TRUE),
                       utils::URLencode(secret, reserved = TRUE))
      response <- xnat_request("GET", path, accept = "application/json")
      result <- xnat_parse_response(response, as = "json")
      isTRUE(result$valid)
    }
    ```

    **3. xnat_invalidate_token() (exported, AUTH-07):**

    ```r
    #' Invalidate an Alias Token
    #'
    #' Invalidates a previously issued alias token so it can no longer be used.
    #'
    #' @param alias Character. The alias portion of the token.
    #' @param secret Character. The secret portion of the token.
    #'
    #' @return Logical. TRUE if invalidation succeeded.
    #' @export
    xnat_invalidate_token <- function(alias, secret) {
      if (missing(alias) || missing(secret)) {
        stop("Both 'alias' and 'secret' must be provided.", call. = FALSE)
      }
      path <- sprintf("/data/services/tokens/invalidate/%s/%s",
                       utils::URLencode(alias, reserved = TRUE),
                       utils::URLencode(secret, reserved = TRUE))
      response <- xnat_request("GET", path)
      httr::status_code(response) == 200
    }
    ```

    Key implementation notes:

    - All functions require prior authentication (xnat_request checks this).
